// Configuração da luz var light = new THREE.PointLight(0xffffff, 1, 100); light.position.set(10, 10, 10); scene.add(light); // Cria o plano verde (chão) var planeGeometry = new THREE.PlaneGeometry(20, 20); var planeMaterial = new THREE.MeshLambertMaterial({ color: 0x00ff00 }); var plane = new THREE.Mesh(planeGeometry, planeMaterial); plane.rotation.x = -Math.PI / 2; scene.add(plane); var trees = []; // Armazena as copas das árvores para animação // Função para criar árvores function criarArvore(x, y, z) { // Cria o tronco da árvore var trunkGeometry = new THREE.CylinderGeometry(0.5, 0.5, 4, 12); var trunkMaterial = new THREE.MeshLambertMaterial({ color: 0x8B4513 }); // Cor marrom para o tronco var trunk = new THREE.Mesh(trunkGeometry, trunkMaterial); trunk.position.set(x, y + 2, z); // Posiciona o tronco // Cria a copa da árvore var crownGeometry = new THREE.ConeGeometry(2, 5, 16); var crownMaterial = new THREE.MeshLambertMaterial({ color: 0x00FF00 }); // Cor verde para a copa var crown = new THREE.Mesh(crownGeometry, crownMaterial); crown.position.set(x, y + 5.5, z); // Ajusta a posição da copa // Adiciona o tronco e a copa à cena scene.add(trunk); scene.add(crown); // Armazena a copa para animação trees.push({ crown, x, y: y + 5.5, z, windOffset: Math.random() * 2 * Math.PI }); // Offset para variar a animação entre árvores } // Cria várias árvores criarArvore(0, 0, 0); criarArvore(-5, 0, 5); criarArvore(5, 0, -5); criarArvore(-5, 0, -5); criarArvore(5, 0, 5); // Variáveis para controlar a animação do vento var windSpeed = 0.002; var windMagnitude = 0.1; // Jogador var playerGeometry = new THREE.SphereGeometry(1, 32, 32); var playerMaterial = new THREE.MeshLambertMaterial({ color: 0xff0000 }); // Cor vermelha para o jogador var player = new THREE.Mesh(playerGeometry, playerMaterial); player.position.set(2, 1, 0); // Posição inicial do jogador scene.add(player); // Movimentação do Jogador var playerSpeed = 0.1; function movePlayer(event) { switch (event.keyCode) { case 87: // W player.position.z -= playerSpeed; break; case 83: // S player.position.z += playerSpeed; break; case 65: // A player.position.x -= playerSpeed; break; case 68: // D player.position.x += playerSpeed; break; } } document.addEventListener('keydown', movePlayer); var cameraFirstPerson = true; function toggleCameraMode() { cameraFirstPerson = !cameraFirstPerson; } document.addEventListener('keydown', function (event) { if (event.keyCode === 67) { toggleCameraMode(); } movePlayer(event); }); // Adiciona luz ambiente var ambientLight = new THREE.AmbientLight(0xaaaaaa); // Luz neutra inicial scene.add(ambientLight); // Configuração inicial da luz direcional para simular o sol var sunlight = new THREE.DirectionalLight(0xffffff, 1); sunlight.position.set(100, 100, 100); scene.add(sunlight); // Função para atualizar o ciclo de dia e noite var dayDuration = 60000; // Duração do dia em milissegundos (1 minuto) var currentTime = 0; // Tempo atual no ciclo de dia/noite function updateDayCycle() { // Atualiza o tempo atual currentTime += 100; // Incrementa o tempo (ajuste conforme necessário para suavidade) if (currentTime >= dayDuration) currentTime = 0; // Reinicia o ciclo após um dia var fractionOfDay = currentTime / dayDuration; var angle = fractionOfDay * Math.PI * 2; // Converte a fração do dia para um ângulo // Atualiza a posição da luz direcional (sol) sunlight.position.set(Math.sin(angle) * 100, Math.cos(angle) * 100, 0); // Atualiza a cor e a intensidade da luz ambiente baseada no tempo do dia if (fractionOfDay < 0.25 || fractionOfDay > 0.75) { // Noite ambientLight.color.setHex(0x555566); // Luz noturna azulada ambientLight.intensity = 0.3; sunlight.intensity = 0; // O sol não está visível } else { // Dia ambientLight.color.setHex(0xaaaaaa); // Luz diurna neutra ambientLight.intensity = 0.5; sunlight.intensity = 1; // Intensidade total durante o dia } } // Inicializa o ciclo de dia e noite setInterval(updateDayCycle, 100); // Atualiza o ciclo de dia/noite a cada 100ms // Função de Animação function animate() { requestAnimationFrame(animate); if (cameraFirstPerson) { camera.position.x = player.position.x; camera.position.y = player.position.y + 1.6; // altura dos olhos camera.position.z = player.position.z;// camera.lookAt(player.position.x, player.position.y, player.position.z - 5); } else { camera.position.x = player.position.x; camera.position.y = player.position.y + 1.7; // altura camera.position.z = player.position.z + 3; camera.lookAt(player.position.x, player.position.y, player.position.z); } trees.forEach(tree => { var time = Date.now() * windSpeed + tree.windOffset; tree.crown.rotation.z = Math.sin(time) * windMagnitude; }); renderer.render(scene, camera); } animate(); // Redimensionar o renderizador quando a janela for redimensionada window.addEventListener('resize', function () { renderer.setSize(window.innerWidth, window.innerHeight); camera.aspect = window.innerWidth / window.innerHeight; camera.updateProjectionMatrix(); });
